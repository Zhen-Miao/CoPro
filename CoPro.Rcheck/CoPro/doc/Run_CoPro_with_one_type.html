Run CoPro on Organoid data

2026-04-13

Content

This Vignette contains an example to run CoPro on an organoid data. Here, we focus on detecting the spatial pattern of the intestinal organoid. Here, we aim to show the unsupervised detection of cell self-organizations.

load packages

library("CoPro")
library_paths <- c(
  "fields", "ggplot2", "tictoc", "irlba",
  "Matrix", "ggsci", "ape", "DESeq2","SummarizedExperiment"
)
# Load required libraries
lapply(library_paths, library, character.only = TRUE)

set some parameters

We can change the number of PCs. For this dataset, this parameter does not affect the results much.

n_pca = 30
cell_types <- c('Epithelial')
n_cell_types <- length(cell_types)
num_bins_x <- 10
num_bins_y <- 10

Load data

Here we load the normalized gene matrix as well as the meta data. The cell type label as well as cell location information are in the meta data.

loc <- "/Users/zhenmiao/Library/CloudStorage/Dropbox/DIALOGUE_plus project/Raj_lab_data/72hr/roi1/"

ra <- read.csv(paste0(loc, 'output/cell_by_gene/cell_by_gene.csv'))
ra <- ra[,2:ncol(ra)]
meta_cell <- read.csv(paste0(loc,'output/attributes/cell_attributes.csv'))

meta_cell$label <- paste0("cell_", meta_cell$label)
rownames(ra) <- meta_cell$label
rownames(meta_cell) <- meta_cell$label

preprocess data

filter expression values

Here, we filter outliers by capping at 95% persentiles

## filter expression outliers
filter_out <- function(x) {
  upper <- quantile(x, 0.95)
  x[x > upper] <- upper
  return(x)
}
ra <- apply(ra, MARGIN = 2, FUN = filter_out)

Normalize data

We then normalize the data by DEseq2. The reasons that we choose DEseq2 instead of total read counts are: (1) All cells can be considered as the same cell type, then we assume that the median gene expression values should be similar across all cells. (2) The number of genes measured are limited (~150), so the total read count is not a good proxy of actual depth of the cells.

## normalize data by DEseq2
counts_matrix <- t(ra) +1 ## count +1, as DEseq2 will need to take log
mode(counts_matrix) <- "integer"

condition <- factor(rep("condition", ncol(counts_matrix)))
colData <- data.frame(row.names = colnames(counts_matrix),
                      condition = condition)

# Create DESeqDataSet object
dds <- DESeqDataSetFromMatrix(countData = counts_matrix,
                              colData = colData,
                              design = ~ 1)
dds <- estimateSizeFactors(dds)
normalized_counts2 <- t(ra) / sizeFactors(dds)
ra_norm <- t(log1p(normalized_counts2))

show the in situ plot

ggplot(meta_cell) +
  geom_point(aes(x = center_x, y = center_y), color = "steelblue", size = 0.8) +
  ggtitle("72h culture, ROI-1") +
  theme_classic()

load data into object

Here, we prepare the four inputs for our pipeline: normalizedData, the normalized expression matrix. locationData, a matrix or data.frame with two columns, x and y to mark the location of each cell. metaData, a data.frame that stores the metadata for each cell.

cellTypes, a vector of character including the cell type label for each cell.

location_data <- meta_cell[, c("center_x", "center_y")]
location_data$x <- location_data$center_x / 5000
location_data$y <- location_data$center_y / 5000

location_data$x_bin <- cut(location_data$x, breaks = num_bins_x, labels = FALSE)
location_data$y_bin <- cut(location_data$y, breaks = num_bins_y, labels = FALSE)
meta_cell$"Celltype" <- "Epithelial"

cell_type_data <- meta_cell[, "Celltype"]

etr <- newCoProSingle(
  normalizedData = ra_norm,
  locationData = location_data,
  metaData = meta_cell,
  cellTypes = cell_type_data
) ## etr stands for enteroid
cell_types_single <- "Epithelial"
etr <- subsetData(etr, cellTypesOfInterest = cell_types_single)

run CoPro pipeline

compute PCA and distances

etr <- computePCA(etr, nPCA = n_pca, center = TRUE, scale. = TRUE)
## By default, we normalize the distance
etr <- computeDistance(etr, distType = "Euclidean2D", normalizeDistance = FALSE)

sigma_choice <- c(0.01, 0.02, 0.05, 0.1, 0.15, 0.2)

## compute kernel. Note that it is a different function for one cell type model.
etr <- computeKernelMatrix(etr, sigmaValues = sigma_choice,
                              upperQuantile = 0.85,
                              normalizeKernel = FALSE, lowerLimit = 5e-7)

run skrCCA and obtain correlations

etr <- runSkrCCA(etr, scalePCs = TRUE, maxIter = 500, nCC = 4)
etr <- computeNormalizedCorrelation(etr, tol = 1e-3)

compute gene and cell scores

etr <- computeGeneAndCellScores(etr)

Look across different sigma values

Here, we hope to pick the sigma value with the highest normalized correlation value. We visualize the trend using ggplot2

ncorr <- getNormCorr(etr)

ggplot(data = ncorr, aes(
  x = sigmaValues,
  y = normalizedCorrelation, group = 1
)) +
  geom_point() +
  geom_line() +
  facet_wrap(facets = ~ ct12 + CC_index,
             labeller = purrr::partial(label_both, sep = " = ")) +
  xlab("sigmaValues") +
  ylab("Norm. Corr.") +
  ggtitle(label = "Norm. Corr. across sigma values", subtitle = "Epi-Epi") +
  theme_minimal()

From the figure above, the highest correlation is when sigma2 = 0.02, so we use this sigma value for the downstream analyses

Cell scores correlation

We can visualize the cell scores with respect to one another by plotting XAKAB against XB, where XA represents the cell scores of each cell in cell type A, and XB represents the cell scores of each cell in cell type B, and KAB represents the Kernel matrix between cell type A and B.

cell_A <- cell_types_single
df_crtt <- getCorrOneType(etr,
                          sigmaValueChoice = 0.1,
                          cellTypeA = cell_A, ccIndex = 1)

ggplot(df_crtt) +
  geom_point(aes(x = AK, y = B), size = 0.5) +
  ggtitle(paste("Correlation plot between", cell_A, "and", cell_A),
          subtitle = paste("sigma_value = ", 0.1)) +
  xlab(paste(cell_A, "%*% Kernal_AB")) +
  ylab(cell_A) +
  theme_minimal()
